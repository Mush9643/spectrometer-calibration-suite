                        from PyQt6.QtWidgets import QDialog, QVBoxLayout, QLabel, QLineEdit, QPushButton, QFormLayout, QApplication
from PyQt6.QtGui import QIcon
from PyQt6.QtCore import QRect, QSize

class SettingsDialog(QDialog):
    def __init__(self, parent=None):
        super().__init__(parent)
        self.setWindowTitle("Настройки подключения")
        self.setWindowIcon(QIcon("M-Photoroom.png"))
        self.setFixedSize(300, 250)  # Фиксированный размер окна

        # Создаем форму для ввода параметров
        layout = QFormLayout()

        # Параметры подключения
        self.port_edit = QLineEdit("COM3")
        self.baud_rate_edit = QLineEdit("115200")
        self.parity_edit = QLineEdit("N")
        self.stop_bits_edit = QLineEdit("1")
        self.byte_size_edit = QLineEdit("8")
        self.timeout_edit = QLineEdit("3")

        # Добавляем поля в форму
        layout.addRow(QLabel("Порт:"), self.port_edit)
        layout.addRow(QLabel("Скорость (baud):"), self.baud_rate_edit)
        layout.addRow(QLabel("Четность (P/N/E):"), self.parity_edit)
        layout.addRow(QLabel("Стоп-биты (1/2):"), self.stop_bits_edit)
        layout.addRow(QLabel("Размер байта (5-8):"), self.byte_size_edit)
        layout.addRow(QLabel("Таймаут (сек):"), self.timeout_edit)

        # Кнопки OK и Cancel
        button_layout = QVBoxLayout()
        ok_button = QPushButton("OK")
        cancel_button = QPushButton("Cancel")
        ok_button.clicked.connect(self.accept)
        cancel_button.clicked.connect(self.reject)
        button_layout.addWidget(ok_button)
        button_layout.addWidget(cancel_button)

        # Добавляем кнопки к основному layout
        main_layout = QVBoxLayout()
        main_layout.addLayout(layout)
        main_layout.addLayout(button_layout)
        self.setLayout(main_layout)

        # Центрируем окно на экране
        self.center()

    def center(self):
        """
        Центрирует окно на экране.
        """
        screen_geometry = QApplication.primaryScreen().geometry()  # Получаем геометрию экрана
        center_point = screen_geometry.center()  # Центр экрана
        frame_geometry = self.frameGeometry()  # Геометрия окна
        frame_geometry.moveCenter(center_point)  # Перемещаем центр окна в центр экрана
        self.move(frame_geometry.topLeft())  # Перемещаем окно в вычисленную позицию

    def get_settings(self):
        """
        Возвращает параметры подключения, введённые пользователем.
        """
        return {
            "port": self.port_edit.text(),
            "baud_rate": int(self.baud_rate_edit.text()),
            "parity": self.parity_edit.text(),
            "stop_bits": int(self.stop_bits_edit.text()),
            "byte_size": int(self.byte_size_edit.text()),
            "timeout": int(self.timeout_edit.text())
        }

# Пример использования
if __name__ == "__main__":
    import sys
    app = QApplication(sys.argv)
    dialog = SettingsDialog()
    if dialog.exec() == QDialog.DialogCode.Accepted:
        settings = dialog.get_settings()
        print("Настройки подключения:", settings)
    sys.exit(app.exec())
